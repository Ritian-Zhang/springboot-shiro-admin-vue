	<script type="text/javascript">
        function submitForm(obj){
            $(obj).jqLoading();
			$('#ff').form('submit', {
				url: "${ctx}/#url_path#/update",
				onSubmit: function(){
					var isValid = $(this).form('enableValidation').form('validate');
					if (!isValid){
                        $(obj).jqLoading("destroy");
					}
					return isValid;	// 返回false终止表单提交
				},
				success: function(data){
                    $(obj).jqLoading("destroy");
					var data = eval('(' + data + ')');  
			        if (data.success){
			        	$("#win").window("close");
			        	grid_.datagrid('reload');//添加成功重新加载数据
			        } else {
			        	$.messager.alert('提示',data.msg,'error');
			        }
				}
			});
			return false;
		}
        function closeForm(){
            $("#win").window("close");
        }
	</script>
